# OpenCV with contrib modules — used for camera capture, frame
# preprocessing, and the JPEG/MJPEG render path. We previously also
# used cv2.aruco for AprilTag detection, but cv2.aruco is a *port* of
# the upstream AprilTag algorithm and is meaningfully worse at
# distance: it loses tags 5-10 ft sooner than WPILib's wrapper around
# the C++ AprilTag library does. We now prefer robotpy-apriltag
# (same lib PhotonVision uses) and fall back to cv2.aruco only if it
# can't be imported.

# Pupil Labs' wrapper around the same upstream C++ AprilTag library
# that PhotonVision / robotpy-apriltag use — identical detection
# algorithm. We use this instead of robotpy-apriltag because:
#   - robotpy-apriltag's aarch64 wheels target manylinux_2_35 (glibc
#     2.35, Bookworm-era), so on Debian Bullseye + Orange Pi 5 stock
#     image (glibc 2.31) pip reports "no matching distribution".
#   - pupil-apriltags ships manylinux2014 wheels (glibc 2.17+) that
#     install cleanly on every supported Pi/Orange Pi image we've
#     tried — Bullseye, Bookworm, Trixie.
# Same algorithm, same long-range performance.
pupil-apriltags>=1.0.4
